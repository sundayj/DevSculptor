### Code blocks

```js
const testArray = [
  "One", "Two", "Three", "Two", "Four",
  "Five", "Four","Three", "One", "Five",
  "Five", "Three", "Four", "One", "Two"
];
const delDuplicates = array => [...new Set(array)];
const result = delDuplicates(testArray);
console.log(result);
// Output:
// "One", "Two", "Three", "Four", "Five"
```
